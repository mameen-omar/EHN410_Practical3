85  // generate primes half the size of the number of bits required for the key
86  getPrime(rsaCtx->p, rsaCtx->numBits/2);
87  getPrime(rsaCtx->q, rsaCtx->numBits/2);
88 
89  mpz_mul(rsaCtx->n, rsaCtx->p, rsaCtx->q); // n = pq

91  mpz_t qTemp;
92  mpz_t pTemp;
93  mpz_init(qTemp); // used for q-1
94  mpz_init(pTemp); // used for p-1
95  mpz_sub_ui(qTemp, rsaCtx->q, 1); // q-1
96  mpz_sub_ui(pTemp, rsaCtx->p, 1); // p-1
97 
98  mpz_mul(rsaCtx->qn, qTemp, pTemp); // q(n) = (p-1)(q-1)
99  mpz_clear(qTemp);
100  mpz_clear(pTemp);
101  mpz_set_ui(rsaCtx->e, CONSTANTE); // set static as per practical spec
102  mpz_invert(rsaCtx->d,rsaCtx->e,rsaCtx->qn); // calc d
103 
104  if(mpz_sgn(rsaCtx->d) < 0) {
105  mpz_add(rsaCtx->d, rsaCtx->d,rsaCtx->qn);
106  }
107  // set private and public
108  mpz_set(rsaCtx->KU[0], rsaCtx->e);
109  mpz_set(rsaCtx->KU[1], rsaCtx->n);
110  mpz_set(rsaCtx->KR[0], rsaCtx->d);
111  mpz_set(rsaCtx->KR[1], rsaCtx->n);
